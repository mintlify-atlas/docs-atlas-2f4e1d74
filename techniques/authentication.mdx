Authentication verifies the identity of a user — confirming *who* they are. NestJS integrates with [Passport.js](https://www.passportjs.org/), the most widely-used Node.js authentication library, through the `@nestjs/passport` package. Passport handles the strategy pattern for pluggable authentication mechanisms, while NestJS wraps it into the familiar Guards and decorators system.

<Note>
`@nestjs/passport` is a separate package and is not included in the NestJS core. You must install it alongside the Passport strategy packages you need.
</Note>

## Local strategy (username and password)

The local strategy authenticates users with a username and password. You implement it by extending `PassportStrategy` from `@nestjs/passport` and calling the `super()` with strategy-specific options.

```typescript auth/local.strategy.ts
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { Strategy } from 'passport-local';
import { AuthService } from './auth.service';

@Injectable()
export class LocalStrategy extends PassportStrategy(Strategy) {
  constructor(private authService: AuthService) {
    super(); // passport-local defaults to { usernameField: 'username' }
  }

  async validate(username: string, password: string): Promise<any> {
    const user = await this.authService.validateUser(username, password);
    if (!user) {
      throw new UnauthorizedException();
    }
    return user;
  }
}
```

Passport calls the `validate()` method automatically after extracting credentials. Whatever you return from `validate()` is attached to the request as `req.user`.

The `AuthService` that checks the user's credentials:

```typescript auth/auth.service.ts
import { Injectable } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { UsersService } from '../users/users.service';

@Injectable()
export class AuthService {
  constructor(
    private usersService: UsersService,
    private jwtService: JwtService,
  ) {}

  async validateUser(username: string, password: string): Promise<any> {
    const user = await this.usersService.findOne(username);
    if (user && user.password === password) {
      const { password, ...result } = user;
      return result;
    }
    return null;
  }

  async login(user: any) {
    const payload = { username: user.username, sub: user.userId };
    return {
      access_token: this.jwtService.sign(payload),
    };
  }
}
```

<Warning>
Never store plain-text passwords. Use a library like `bcrypt` to hash passwords before storing them and to compare during login.
</Warning>

## JWT strategy

After a user logs in, your application issues a signed JWT. Subsequent requests include the token in the `Authorization: Bearer <token>` header, and the JWT strategy validates it.

```typescript auth/jwt.strategy.ts
import { Injectable } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor() {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      ignoreExpiration: false,
      secretOrKey: process.env.JWT_SECRET,
    });
  }

  async validate(payload: any) {
    return { userId: payload.sub, username: payload.username };
  }
}
```

Passport verifies the JWT signature and expiry before calling `validate()`. The returned object is attached to `req.user`.

## Creating the AuthModule

<Steps>
  <Step title="Register JwtModule and strategies">
    ```typescript auth/auth.module.ts
    import { Module } from '@nestjs/common';
    import { JwtModule } from '@nestjs/jwt';
    import { PassportModule } from '@nestjs/passport';
    import { AuthService } from './auth.service';
    import { AuthController } from './auth.controller';
    import { JwtStrategy } from './jwt.strategy';
    import { LocalStrategy } from './local.strategy';
    import { UsersModule } from '../users/users.module';

    @Module({
      imports: [
        UsersModule,
        PassportModule,
        JwtModule.register({
          secret: process.env.JWT_SECRET,
          signOptions: { expiresIn: '1h' },
        }),
      ],
      providers: [AuthService, LocalStrategy, JwtStrategy],
      controllers: [AuthController],
    })
    export class AuthModule {}
    ```
  </Step>
  <Step title="Add the login endpoint">
    The login route uses the `LocalAuthGuard`, which runs the local strategy before the handler:

    ```typescript auth/auth.controller.ts
    import { Controller, Post, UseGuards, Request } from '@nestjs/common';
    import { AuthService } from './auth.service';
    import { LocalAuthGuard } from './local-auth.guard';

    @Controller('auth')
    export class AuthController {
      constructor(private authService: AuthService) {}

      @UseGuards(LocalAuthGuard)
      @Post('login')
      async login(@Request() req) {
        return this.authService.login(req.user);
      }
    }
    ```
  </Step>
</Steps>

## Guards

Create thin guard classes that delegate to the Passport strategy:

```typescript auth/local-auth.guard.ts
import { Injectable } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';

@Injectable()
export class LocalAuthGuard extends AuthGuard('local') {}
```

```typescript auth/jwt-auth.guard.ts
import { Injectable } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {}
```

## Protecting routes

Apply `JwtAuthGuard` to any route or controller that requires an authenticated user:

```typescript profile/profile.controller.ts
import { Controller, Get, UseGuards, Request } from '@nestjs/common';
import { JwtAuthGuard } from '../auth/jwt-auth.guard';

@Controller('profile')
export class ProfileController {
  @UseGuards(JwtAuthGuard)
  @Get()
  getProfile(@Request() req) {
    return req.user;
  }
}
```

To protect all routes in your application globally, register the guard in `AppModule`:

```typescript app.module.ts
import { Module } from '@nestjs/common';
import { APP_GUARD } from '@nestjs/core';
import { JwtAuthGuard } from './auth/jwt-auth.guard';

@Module({
  providers: [
    {
      provide: APP_GUARD,
      useClass: JwtAuthGuard,
    },
  ],
})
export class AppModule {}
```

<Tip>
When you register `JwtAuthGuard` globally, create a `@Public()` decorator using `SetMetadata` and check for it in the guard's `canActivate()` method to opt specific routes out of authentication.
</Tip>

## Async JwtModule configuration

In production you often read the JWT secret from a `ConfigService`. Use `JwtModule.registerAsync()` for this:

```typescript auth/auth.module.ts
import { Module } from '@nestjs/common';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { JwtModule } from '@nestjs/jwt';
import { PassportModule } from '@nestjs/passport';

@Module({
  imports: [
    ConfigModule,
    PassportModule,
    JwtModule.registerAsync({
      imports: [ConfigModule],
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        secret: config.get<string>('JWT_SECRET'),
        signOptions: { expiresIn: '1h' },
      }),
    }),
  ],
})
export class AuthModule {}
```
